4.1.4 Optimize

Basing on the Pros&Cons analysis above, we know that Gradient Descent is a unstable algorithm. And we could improve that a little bit. This chapter aims to introduce some practical ways to imporve gradient descent.

4.1.4.1 Feature Scaling
4.1.4.1.1 Why Feature Scaling

Considering different features have way different range, the curve of the Cost Function will be “flat”, say, oval shape (We prefer circle). In this case, the fitting process takes much longer than usual. By feature scaling, the data range will be pretty close, so that the curve of Cost Function is close to circle again.

4.1.4.2 Learning Rate

Another reason causes Gradient Descent unstable is the learning rate.

  • if the is too small : small convergence
  • if the is too large : may not decrease on every iteration --> may not converge

To choose \alpha, try to debug with values with multiple 3.

E.g., … 0.001, 0.003, 0.010, 0.030, 0.100, 0.300, 1.0, …

%% Initialization

clear ; close all; clc

%% =================== 1. Prepare Data ===================

fprintf('=================== 1. Prepare Data ===================\n');

 

% Load data.

% Get the data from this link: https://raw.githubusercontent.com/JasonDean-1/Machine-Learning-Training-Sets/master/Linear%20Model/data1.txt

data = load('data1.txt');

 

% Split the data into X and Y matrix.

X = data(:, 1);

Y = data(:, 2);

 

% Get the length of the data -- Y is a vector

len = length(Y); % number of training examples

 

%% =================== 2. Plot Data ===================

fprintf('=================== 2. Plot Data ===================\n');

% Plot Data -- The data have only one attribute --> See more of plot two. See more at

% https://www.gnu.org/software/octave/doc/interpreter/Plotting.html#Plotting

 

% open a new figure window

figure;

% Plot the data

plot(X, Y, 'rx', 'MarkerSize', 10);

% Set X and Y axis label

ylabel('Profit in $10,000s');

xlabel('Population of City in 10,000s');

 

%% =================== 3. Gradient Descent ===================

fprintf('=================== 3. Gradient Descent ===================\n');

% Add a column(x0) of ones to X

 

X = [ones(len, 1), X];

theta = zeros(2, 1);

alpha = 0.01;

ITERATION = 1500;

jTheta = zeros(ITERATION, 1);

 

for iter = 1:ITERATION

    % Perform a single gradient descent on the parameter vector

    % Note: since the theta will be updated, a tempTheta is needed to store the data.

    tempTheta = theta;

    theta(1) = theta(1) - (alpha / len) * (sum(X * tempTheta - Y));  % ignore the X(:,1) since the values are all ones.

    theta(2) = theta(2) - (alpha / len) * (sum((X * tempTheta - Y) .* X(:,2)));

 

    %% =================== 3.1 Compute Cost ===================

    jTheta(iter) = sum((X * theta - Y) .^ 2) / (2 * len);

endfor

 

% Print theta to screen

fprintf('Theta updated by gradient descent: %f %f \n', theta(1), theta(2));

 

% Print jTheta

jTheta

 

 

%% =================== 4. Plot Linear Fit Linear ===================

fprintf('=================== 4. Plot Linear Fit Linear ===================\n');

% keep previous plot visible

hold on;

plot(X(:,2), X*theta, '-')

legend('Training Data', 'Linear Regression')

% Don't overlay any more plots on this figure

hold off

 

%% =================== 5. Try to Predict ===================

predict1 = [1, 3.5] *theta;

fprintf('For population = 35,000, we predict a profit of %f\n', predict1*10000);

predict2 = [1, 7] * theta;

fprintf('For population = 70,000, we predict a profit of %f\n', predict2*10000);

 

 

%% =================== 5. Visualizing J(theta_0, theta_1) ===================

fprintf('===================5. Visualizing J(theta_0, theta_1) ===================\n');

% Grid over which we will calculate J

theta0Vals = linspace(-10, 10, 100);

theta1Vals = linspace(-1, 4, 100);

 

% initialize jVal to a matrix of 0's

jVal = zeros(length(theta0Vals), length(theta1Vals));

 

% Fill out jVal

for i = 1:length(theta0Vals)

    for j = 1:length(theta1Vals)

	  thetaVector = [theta0Vals(i); theta1Vals(j)];

      %% =================== 5.1 Compute Cost ===================

      jVal(i,j) = sum((X * thetaVector - Y) .^ 2) / (2 * len);

    end

end

 

% Because of the way meshgrids work in the surf command, we need to

% transpose jVal before calling surf, or else the axes will be flipped

jVal = jVal';

% Surface plot

figure;

surf(theta0Vals, theta1Vals, jVal)

xlabel('\theta_0');

ylabel('\theta_1');

 

% Contour plot

figure;

% Plot jVal as 15 contours spaced logarithmically between 0.01 and 100

contour(theta0Vals, theta1Vals, jVal, logspace(-2, 3, 20))

xlabel('\theta_0');

ylabel('\theta_1');

hold on;

plot(theta(1), theta(2), 'rx', 'MarkerSize', 10, 'LineWidth', 2);

5.1.2 Example 2 - Multi-feature Linear Classification

%% Initialization

clear ; close all; clc

%% =================== 1. Prepare Data ===================

fprintf('=================== 1. Prepare Data ===================\n');

 

% Load data.

% Get the data from this link: https://raw.githubusercontent.com/JasonDean-1/Machine-Learning-Training-Sets/master/Linear%20Model/data2.txt

data = load('ex1data2.txt');

 

% Split the data into X and Y matrix.

X = data(:, 1:2);

Y = data(:, 3);

 

% Get the length of the data -- Y is a vector

len = length(Y); % number of training examples

 

%% =================== 2. Feature Scaling ===================

fprintf('=================== 2. Feature Scaling ===================\n');

% Initialize parameters

mu = zeros(1, size(X, 2));

sigma = zeros(1, size(X, 2));

% Scale features and set them to zero mean

mu = mean(X);

sigma = std(X);

xScaled = (X - repmat(mu, size(X, 1), 1)) ./ (repmat(sigma, size(X, 1), 1));

 

% Add a column(x0) of ones to X

X = [ones(len, 1), xScaled];

%% =================== 3. Gradient Descent ===================

fprintf('=================== 3. Gradient Descent ===================\n');

 

theta = zeros(size(X, 2), 1);

alpha = 0.01;

ITERATION = 8000;

jTheta = zeros(ITERATION, 1);

 

for iter = 1:ITERATION

    % Perform a single gradient descent on the parameter vector

    theta = theta - alpha / len * X' * (X * theta - Y);

 

    %% =================== 3.1 Compute Cost ===================

    jTheta(iter) = sum((X * theta - Y) .^ 2) / (2 * len);

endfor

 

% Print theta to screen

fprintf('Theta updated by gradient descent: %f %f \n', theta(1), theta(2));

 

%% =================== 3.2 Plot the Convergence Graph ===================

% Plot the convergence graph

figure;

plot(1:numel(jTheta), jTheta, '-b', 'LineWidth', 2);

xlabel('Number of iterations');

ylabel('Cost J');

 

% Display gradient descent's result

fprintf('Theta computed from gradient descent: \n');

fprintf(' %f \n', theta);

fprintf('\n');

 

%% =================== 3.3 Predict the Price ===================

price = [1 (([2000 5]-mu) ./ sigma)] * theta ;

% ============================================================

 

fprintf(['Predicted price of a 2000 sq-ft, 5 br house ' ...

         '(using gradient descent):\n $%f\n'], price);

 

%% =================== 4. Normal Equations ===================

fprintf('=================== 4. Normal Equations ===================\n');

 

%% Reload Data

data = csvread('ex1data2.txt');

X = data(:, 1:2);

Y = data(:, 3);

len = length(Y);

 

% Add intercept term to X

X = [ones(len, 1) X];

theta = zeros(size(X, 2), 1);

 

% Calculate the parameters from the normal equation

theta = pinv( X' * X ) * X' * Y;

 

% Display normal equation's result

fprintf('Theta computed from the normal equations: \n');

fprintf(' %f \n', theta);

fprintf('\n');

 

% Estimate the price of a 2000 sq-ft, 5 br house

price = [1 2000 5] * theta ;

 

% ============================================================

 

fprintf(['Predicted price of a 2000 sq-ft, 5 br house (using normal equations):\n $%f\n'], price);
